


pass length = 12, change the number to have a longer or shorter pass

# Characters only -- Example: TPBahbBuHLqL
cat /dev/urandom | tr -dc '[:alpha:]' | head -c${1:-12}; echo;

# Numbers only -- Example: 767933767552
cat /dev/urandom | tr -dc '[:digit:]' | head -c${1:-12}; echo;


# Numeric alphas -- Example: zVVRIsEUfFX2
cat /dev/urandom | tr -dc '[:alnum:]' | head -c${1:-12}; echo;
