-- Vers: 1.3

-- Replace xxx with USER to generate a standard script.
-- You only need to plan to change the name of the DATA diskgroup and the 2 user passwords
-- with Vault create user from dvmanager user 


set echo on

-- TABLESPACES DATA
CREATE BIGFILE TABLESPACE xxx_DATA
    DATAFILE '+DATA' SIZE 1G AUTOEXTEND ON NEXT 1G MAXSIZE 20G
    DEFAULT NOCOMPRESS NO INMEMORY
    ONLINE
    SEGMENT SPACE MANAGEMENT AUTO
    EXTENT MANAGEMENT LOCAL AUTOALLOCATE;

-- TABLESPACES DATA
CREATE BIGFILE TABLESPACE xxx_INDX
    DATAFILE '+DATA' SIZE 1G AUTOEXTEND ON NEXT 1G MAXSIZE 10G
    DEFAULT NOCOMPRESS NO INMEMORY
    ONLINE
    SEGMENT SPACE MANAGEMENT AUTO
    EXTENT MANAGEMENT LOCAL AUTOALLOCATE;


-- Creation of READONLY and READWRITE roles
CREATE ROLE             xxx_RO NOT IDENTIFIED;
CREATE ROLE             xxx_RW NOT IDENTIFIED;


-- OBJ User Creation
CREATE USER             xxx_OBJ
IDENTIFIED BY           "&Pass"
DEFAULT TABLESPACE      xxx_DATA
TEMPORARY TABLESPACE TEMP;

ALTER USER              xxx_OBJ QUOTA UNLIMITED ON xxx_DATA;
ALTER USER              xxx_OBJ QUOTA UNLIMITED ON xxx_INDX;

-- The OBJ user must not have the CONNECT role and CREATE SESSION privilege
GRANT       RESOURCE TO xxx_OBJ;


-- Creation of SV User
CREATE USER             xxx_SV
IDENTIFIED BY           "&Pass"
DEFAULT TABLESPACE      xxx_DATA
TEMPORARY TABLESPACE TEMP;

GRANT CONNECT        TO xxx_SV;
GRANT CREATE SESSION TO xxx_SV;
GRANT xxx_RW         TO xxx_SV;
GRANT CREATE SYNONYM TO xxx_SV;


-- Vault
EXEC DBMS_MACADM.ADD_AUTH_TO_REALM ('Realm TOTALE','&USERNAME');
EXEC DBMS_MACADM.ADD_AUTH_TO_REALM ('Realm PAN','&USERNAME');
EXEC DBMS_MACADM.ADD_AUTH_TO_REALM ('Realm MOV','&USERNAME');
EXEC DBMS_MACADM.ADD_AUTH_TO_REALM ('Realm ANAG','&USERNAME');
