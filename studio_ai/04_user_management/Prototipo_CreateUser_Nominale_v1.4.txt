-- Vers: 1.3
-- Le utenze nominali hanno il formato COxxxx per esterni e D0xxxx per gli interni e vengono comunicati da Security Governance
-- Controllare il campo Attività del Hp Service Manager

#####Creazione Profile single pluggable DB#####
create profile PROFILO_USER limit
                 idle_time  15
     failed_login_attempts   5
        password_life_time  80
       password_reuse_time 450
        password_reuse_max   5
        password_lock_time  30
       password_grace_time  10
  password_verify_function  VERIFY_FUNCTION;


#####Creazione Profile GLOBAL pluggable DB#####
create profile C##PROFILO_USER limit
                 idle_time  15
     failed_login_attempts   5
        password_life_time  80
       password_reuse_time 450
        password_reuse_max   5
        password_lock_time  30
       password_grace_time  10
  password_verify_function  VERIFY_FUNCTION
  container=ALL;


## Creazione utenza nominale su single pluggable con VAULT attivo: 
conn dvmanager/&pass
undefine Utente
accept Utente prompt "Inserire nome Utente: " 
CREATE USER &&Utente
        IDENTIFIED BY "Cambiami1!"
   DEFAULT TABLESPACE USERS
 TEMPORARY TABLESPACE TEMP
              PROFILE PROFILO_USER
 ACCOUNT UNLOCK
 PASSWORD EXPIRE;

GRANT        CONNECT TO &&Utente;
GRANT CREATE SESSION TO &&Utente;
ALTER USER              &&Utente DEFAULT ROLE ALL; 

conn &DBA/&pass
GRANT CREATE SYNONYM TO &&Utente;
GRANT VAULT_TOTALE_PARTICIPANT to &&Utente;



## Creazione utenza nominale su pluggable senza VAULT: 
undefine Utente
accept Utente prompt "Inserire nome Utente: " 
CREATE USER &&Utente IDENTIFIED BY "Cambiami1!" DEFAULT TABLESPACE USERS TEMPORARY TABLESPACE TEMP PROFILE C##PROFILO_USER ACCOUNT UNLOCK PASSWORD EXPIRE;          

GRANT        CONNECT TO &&Utente;
GRANT CREATE SESSION TO &&Utente;
ALTER USER              &&Utente DEFAULT ROLE ALL; 

conn &DBA/&pass
GRANT CREATE SYNONYM TO &&Utente;
GRANT VAULT_TOTALE_PARTICIPANT to &&Utente;
